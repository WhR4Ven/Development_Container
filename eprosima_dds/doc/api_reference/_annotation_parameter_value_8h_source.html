704  RTPS_DllAPI void from_string(
705  const std::string& value)
706  {
707  switch (m__d)
708  {
709  case TK_BOOLEAN:
710  {
711  std::string val_ = value;
712  std::transform(val_.begin(), val_.end(), val_.begin(),
713  [](unsigned char c)
714  {
715  return static_cast<char>(std::tolower(c));
716  });
717  boolean_value(val_.compare("0") != 0 || val_.compare(CONST_TRUE) == 0);
718  }
719  break;
720  case TK_BYTE:
721  {
722  byte_value(static_cast<uint8_t>(std::stoul(value)));
723  }
724  break;
725  case TK_INT16:
726  {
727  int16_value(static_cast<int16_t>(std::stoi(value)));
728  }
729  break;
730  case TK_INT32:
731  {
732  int32_value(static_cast<int32_t>(std::stoi(value)));
733  }
734  break;
735  case TK_INT64:
736  {
737  int64_value(static_cast<int64_t>(std::stoll(value)));
738  }
739  break;
740  case TK_UINT16:
741  {
742  uint_16_value(static_cast<uint16_t>(std::stoul(value)));
743  }
744  break;
745  case TK_UINT32:
746  {
747  uint32_value(static_cast<uint32_t>(std::stoul(value)));
748  }
749  break;
750  case TK_UINT64:
751  {
752  uint64_value(static_cast<uint64_t>(std::stoull(value)));
753  }
754  break;
755  case TK_FLOAT32:
756  {
757  float32_value(std::stof(value));
758  }
759  break;
760  case TK_FLOAT64:
761  {
762  float64_value(std::stod(value));
763  }
764  break;
765  case TK_FLOAT128:
766  {
767  float128_value(std::stold(value));

769  break;
770  case TK_CHAR8:

796  break;
797  default:

799  }
